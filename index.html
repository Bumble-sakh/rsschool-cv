    function customHttp() {
      return {
        get(url, cb) {
          try {
            const xhr = new XMLHttpRequest();
            xhr.open('GET', url);
            xhr.send();

            xhr.addEventListener('load', () => {
              if(Math.floor(xhr.status / 100) !== 2) {
                cb(`Error. Status code: ${xhr.status}`, xhr);
                return ;
              }
              const response = JSON.parse(xhr.responseText);
              cb(null, response);
            });

            xhr.addEventListener('error', () => {
              cb(`Error. Status code: ${xhr.status}`, xhr);
            });
          } catch (error) {
            cb(`Error. ${error}`);
          }
        },
      }
    }
